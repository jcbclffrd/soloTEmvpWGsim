━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  TE sequences are repetitive by definition — the same ~300 bp Alu or L1
  sequence appears in thousands of locations across the genome.  When STAR
  aligns a synthetic read from TE_001, it may find 500 equally valid positions.
  With --outSAMmultNmax 1 --outMultimapperOrder Random, one is chosen at
  random → only ~1/500 reads land back on TE_001, the rest scatter to real
  genome copies of the same repeat.  soloTE then correctly counts UMIs at
  each of those positions, spreading counts across the genome.

  Observed losses this run:
    Reads lost to 'too many loci' filter:  0.00%  (STAR drops reads > 100 alignments)
    Multi-mapping dispersal (estimated):   98%  of remaining reads scattered

  This is not a pipeline error — it is the expected behaviour of aligning
  short reads from repetitive elements to a complete reference genome.
  A real experiment would show the same pattern; the key validation point
  is that ALL planted loci are DETECTED, even if counts are diluted.
